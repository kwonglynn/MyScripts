1. Complete pyplot tutorial:

>>> import numpy as np
>>> import matplotlib
>>> import matplotlib.pyplot as plt

>>> x = np.linspace(start=0, stop=10, num=100) #Return evenly spaced numbers over a specified interval.

>>> fig, ax = plt.subplots(figsize=(12,6))
>>> # Now use the axes object to add stuff to plot
>>> ax.plot(x, x**2, 'r.-', label='line 1')
>>> # The one below is use explicit names for line features.
>>> ax.plot(x, x**3, color='red', alpha=0.5, linewidth=1.0, linestyle='-',
            marker='s', markersize=8, markerfacecolor='green', label='line 2')

>>> # Add a legend to the figure, here at the best position (0).
>>> ax.legend(loc=0)

>>> # Set the axis ranges.
>>> ax.set_xlim([1,10])
>>> ax.set_ylim([1,100])

>>> # Set the labels and titles.
>>> ax.set_xlabel('x', fontsize=16)
>>> ax.set_ylabel('y', fontsize=16)
>>> ax.set_title('title', fontsize=20);

>>> # Set the ticks on the axes.
>>> ax.tick_params(labelsize=15) # Set the tick size on the axes.
>>> ax.set_xticks(np.arange(0,11,2))
>>> ax.set_xticklabels([r'$\alpha$', r'$\beta$', r'$\gamma$', r'$\delta$', r'$\epsilon$'], fontsize=15)
>>> mpl.rcParams['mathtext.default'] = 'regular' # Set the math text to be regular, rather than italic.
>>> yticks = np.arange(0,101, 10)
>>> ax.set_yticks(yticks)
>>> ax.set_yticklabels(["$%.1f$" % y for y in yticks], fontsize=15) # use LaTeX formatted labels

>>> # Set scientific notation for axes with large numbers:
>>> from matplotlib import ticker
>>> formatter = ticker.ScalarFormatter(useMathText=True)
>>> formatter.set_scientific(True)
>>> formatter.set_powerlimits((-1,1))
>>> ax.yaxis.set_major_formatter(formatter)

>>> # Set the distance between x and y axis and the numbers on the axes (tick labels)
>>> matplotlib.rcParams['xtick.major.pad'] = 5
>>> matplotlib.rcParams['ytick.major.pad'] = 5
>>> # Set the padding between axis label and axis numbers
>>> ax.xaxis.labelpad = 5
>>> ax.yaxis.labelpad = 5

>>> # Add axis grid:
>>> ax.grid(True) # Default grid apperance.
>>> ax.grid(color='b', alpha=0.5, linestyle='dashed', linewidth=0.5) # Custom grid appearance.

>>> # Customize axis spines:
>>> ax.spines['bottom'].set_color('blue')
>>> ax.spines['left'].set_linewidth(2)
>>> # Hide the top and right spines.
>>> ax.spines['top'].set_linewidth(0)
>>> ax.spines['right'].set_linewidth(0)

>>> # Text annotaton, label the datapoints:
>>> y = x ** 2
>>> for i in range(0, len(x), 10):
        plt.text(x[i], y[i], str(x[i]), horizontalalignment='left', size='medium', color='black', fontsize=12)

>>> # Adjust the figure so that the lables are not clipped.
>>> fig.subplots_adjust(left=0.15, right=0.9, bottom=0.1, top=0.9);
>>> fig.savefig("filename.png", dpi=600)

2. Draw multiple figures:

>>> import matplotlib.pyplot as plt
>>> import numpy as np

>>> x = np.linspace(0, 10, 100)
>>> y = np.exp(x)

>>> fig, axes = plt.subplots(1, 2, figsize=(10,5))
>>> axes[0].plot(x, y)
>>> axes[1].plot(x, y)
>>> axes[1].set_yscale("log") # Set a logarithmic scale for one ax.

>>> fig.tight_layout() # Used to avoid axes overlap and make the figures more compact.

3. Twin axes:

Sometimes it is useful to have dual y axes in a figure. For example, when plotting curves with different units together.

>>> fig, ax1 = plt.subplots()
>>> x = np.linspace(0, 5, 11)

>>> ax1.plot(x, x**2, lw=2, color="blue")
>>> ax1.set_ylabel(r"area $(m^2)$", fontsize=18, color="blue")
>>> for label in ax1.get_yticklabels():
        label.set_color("blue")

>>> ax2 = ax1.twinx()
>>> ax2.plot(x, x**3, lw=2, color="red")
>>> ax2.set_ylabel(r"volume $(m^3)$", fontsize=18, color="red")
>>> for label in ax2.get_yticklabels():
        label.set_color("red")

4. Customize the axes:

>>> import numpy as np
>>> x = np.linspace(0, 5, 11)
>>> y = x ** 2
>>> fig = plt.figure()

>>> axes1 = fig.add_axes([0.1, 0.1, 0.8, 0.8]) # main axes
>>> axes2 = fig.add_axes([0.2, 0.5, 0.4, 0.3]) # inset axes

>>> # Larger Figure Axes 1
>>> axes1.plot(x, y, 'b')
>>> axes1.set_xlabel('X_label_axes2')
>>> axes1.set_ylabel('Y_label_axes2')
>>> axes1.set_title('Axes 2 Title')

>>> # Insert Figure Axes 2
>>> axes2.plot(y, x, 'r')
>>> axes2.set_xlabel('X_label_axes2')
>>> axes2.set_ylabel('Y_label_axes2')
>>> axes2.set_title('Axes 2 Title');

5. Axes where x and y is zero:

Check here to see how to set the zero of the axes.

>>> import matplotlib.pyplot as plt
>>> import numpy as np
>>> fig, ax = plt.subplots()

>>> ax.spines['right'].set_color('none')
>>> ax.spines['top'].set_color('none')

>>> ax.xaxis.set_ticks_position('bottom')
>>> ax.spines['bottom'].set_position(('data',0)) # set position of x spine to x=0
>>> ax.yaxis.set_ticks_position('left')
>>> ax.spines['left'].set_position(('data',0)) # set position of y spine to y=0

>>> x = np.linspace(-2, 2., 50)
>>> ax.plot(x, x**3);

6. Make a contour:

Contour tutorial

>>> import matplotlib
>>> import numpy as np
>>> import matplotlib.cm as cm
>>> import matplotlib.pyplot as plt

>>> delta = 0.025
>>> x = np.arange(-3.0, 3.0, delta)
>>> y = np.arange(-2.0, 2.0, delta)
>>> X, Y = np.meshgrid(x, y)
>>> Z1 = np.exp(-X**2 - Y**2)
>>> Z2 = np.exp(-(X - 1)**2 - (Y - 1)**2)
>>> Z = (Z1 - Z2) * 2

>>> fig, ax = plt.subplots()
>>> CS = ax.contour(X, Y, Z)
>>> ax.clabel(CS, inline=1, fontsize=10)
>>> ax.set_title('Simple contour with default labels')
